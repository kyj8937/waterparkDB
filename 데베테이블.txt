CREATE TABLE 지점(
지점명 VARCHAR(200) not null,
종일권_대인 INT NOT NULL,
종일권_소인 INT NOT NULL,
오후권_대인 INT NOT NULL,
오후권_소인 INT NOT NULL,
매출번호 varchar(15) NOT NULL,
관리자아이디 VARCHAR(20) NOT NULL,
PRIMARY KEY (지점명)
);


create table 직원(
직원아이디 varchar(20) not null,
직원비밀번호 varchar(20) not null,
직원이름 varchar(50) not null,
직원전화번호 varchar(20) not null,
관리자아이디 varchar(20) not null,
지점명 VARCHAR(200) not null,
primary key(직원아이디)
);


create table 관리자(
관리자아이디 varchar(20) not null,
관리자비밀번호 varchar(20) not null,
지점명 VARCHAR(200) not null,
primary key(관리자아이디)
);

create table 장비(
장비번호 varchar(10) not null,
장비명 varchar(200) not null,
수량 int not null,
대여료 int not null,
사이즈 varchar(10) not null,
대여번호 varchar(10) not null,
직원아이디 varchar(20) not null,
primary key(장비번호)
);

CREATE TABLE 놀이기구(
놀이기구식별번호 VARCHAR(30) NOT NULL,
기구명 varchar(200) not null,
점검일자 date NOT NULL,
운휴일자 date NOT NULL,
이용객수 INT NOT NULL,
지점명 VARCHAR2(200) not null,
직원아이디 varchar(20) not null,
PRIMARY KEY(놀이기구식별번호)
);

CREATE TABLE 카드사(
카드회사명 VARCHAR(150) NOT NULL,
할인혜택 VARCHAR(300) NOT NULL,
지점명 VARCHAR2(200) not null,
PRIMARY KEY(카드회사명)
);

CREATE TABLE 매출(
매출번호 varchar(15) NOT NULL,
총액 INT NOT NULL,
날짜 date NOT NULL,
지점명 VARCHAR2(200) not null,
관리자아이디 varchar(20) not null,
PRIMARY KEY(매출번호)
);
